\section{Comparison}

\begin{frame}{Open Source Prompt vs Spec-Driven Development}
  \begin{block}{Both Use AI for Code Generation}
    \scriptsize
    Different inputs, same goal: AI-assisted software development
  \end{block}

  \begin{block}{Spec-Driven Development (GitHub)}
    \scriptsize
    \begin{itemize}
      \item \textbf{Input}: Formal specifications (structured documents)
      \item \textbf{AI Role}: Interprets specs to generate code
      \item \textbf{Focus}: Upfront planning and specification refinement
      \item \textbf{Workflow}: Specs $\rightarrow$ AI generates code $\rightarrow$ Implementation
    \end{itemize}
  \end{block}

  \begin{block}{Open Source Prompt Initiative}
    \scriptsize
    \begin{itemize}
      \item \textbf{Input}: Conversational prompts (natural language)
      \item \textbf{AI Role}: Responds to prompts to generate code
      \item \textbf{Focus}: Capturing and sharing prompt history
      \item \textbf{Workflow}: Prompts $\rightarrow$ AI generates code $\rightarrow$ Document
    \end{itemize}
  \end{block}

  \begin{block}{Key Insight}
    \scriptsize
    \textbf{Complementary approaches}: Specs provide structure, prompts provide flexibility and reproducibility
  \end{block}
\end{frame}
