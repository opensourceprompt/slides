\section{Introduction}

\begin{frame}{Introduction}
  \begin{itemize}
    \item AI-assisted development is rapidly transforming how we write code
    \item Tools like Claude Code, Gemini, GitHub Copilot are becoming essential
    \item \textbf{Challenge}: How do we ensure AI-generated code is:
    \begin{itemize}
      \item Reproducible
      \item Auditable
      \item Shareable
      \item Adaptable
    \end{itemize}
    \item Current gap: No standardized way to capture and share the prompts that generated the code
  \end{itemize}
  \vfill
  \textit{Open Source Prompt aims to solve this reproducibility challenge.}
\end{frame}
